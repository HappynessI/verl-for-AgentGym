================================================================================
6. vLLM 推理服务 (端口 8000)
================================================================================
# 单 GPU 启动
CUDA_VISIBLE_DEVICES=7 vllm serve /Data/public/Qwen3-1.7B \
    --served-model-name qwen3 \
    --host 0.0.0.0 \
    --port 8000 \
    --max-model-len 16384 \
    --gpu-memory-utilization 0.9

# 验证: curl http://127.0.0.1:8000/v1/models

# 查看当前加载的模型名称和路径:
curl -s http://127.0.0.1:8000/v1/models | python3 -c "import sys,json;d=json.load(sys.stdin)['data'][0];print(f\"Model ID: {d['id']}\nModel Path: {d['root']}\")"

# 停止 vLLM 服务
kill $(ps aux | grep 'vllm serve' | grep -v grep | awk '{print $2}')
